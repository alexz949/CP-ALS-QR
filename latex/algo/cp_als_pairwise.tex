\begin{algorithmic}[1]
    \Function{$\{\M{A}_{k}\}=$ CP-ALS}{$\T{X},r$}
      \State Setup factor matrices  $\mat{A}_1$, $\mat{A}_2$, $\dots$, $\mat{A}_d$
      \State Compute QR decomposition for factor matrices $\mat{Q}_d\mat{R}_d, \dots \mat{Q}_1\mat{R}_1$
      \While{not converge}
      \For{$k = 1, \dots, d$}
        \State $\T{Y} =  \T{X} \times_1 \mat{Q}_1^\top \dots \times_{k-1} \mat{Q}_{k-1}^\top \times_{k+1} \mat{Q}_{k+1}^\top \dots \times_d \mat{Q}_d^\top$ \label{l:Pair-TTM} \Comment{Multi-TTM}
        \State Permute $\T[]{Y}$  
        \State $\mat{R}_0 = \mat{R}_1$ 
        \For{$i = 2, \dots, d$}
        \State $[\mat{Q}_0,\mat{R}_0] = \Call{QR}{\mat{R}_0 \odot \mat{R}_i}$
        \State $\T{Y}= \Call{collapse}{\T{Y}}$
        \State $\T[]{Y} = \T[]{Y} \times_1 \mat{Q}_0$ \Comment{TTM} \label{l:pair-R-TTM}

        \EndFor
              
      \State Solve $  \mat{\hat{A}}_k\mat{R}_0^\top = \mat{Y}_{(k)}$ \Comment{via TRMS}
      
      \State $\mat{A}_k = \mat{Q}_k\mat{R}_k$   \Comment{Update $\mat{Q}_k\mat{R}_k$ and $\mat{A}_k$}    
      \EndFor
      \EndWhile
    \EndFunction
  \end{algorithmic}
